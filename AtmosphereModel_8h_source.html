124 /*
125  * NOTES ON INPUT VARIABLES:
126  * UT, Local Time, and Longitude are used independently in the
127  * model and are not of equal importance for every situation.
128  * For the most physically realistic calculation these three
129  * variables should be consistent (lst=sec/3600 + g_long/15).
130  * The Equation of Time departures from the above formula
131  * for apparent local time can be included if available but
132  * are of minor importance.
133  *
134  * f107 and f107A values used to generate the model correspond
135  * to the 10.7 cm radio flux at the actual distance of the Earth
136  * from the Sun rather than the radio flux at 1 AU. The following
137  * site provides both classes of values:
138  * ftp://ftp.ngdc.noaa.gov/STP/SOLAR_DATA/SOLAR_RADIO/FLUX/
139  *
140  * f107, f107A, and ap effects are neither large nor well
141  * established below 80 km and these parameters should be set to
142  * 150., 150., and 4. respectively.
143  */

151 struct nrlmsise_output {
152  double d[9]; /* densities */
153  double t[2]; /* temperatures */
154 };
155 /*
156  * OUTPUT VARIABLES:
157  * d[0] - HE NUMBER DENSITY(CM-3)
158  * d[1] - O NUMBER DENSITY(CM-3)
159  * d[2] - N2 NUMBER DENSITY(CM-3)
160  * d[3] - O2 NUMBER DENSITY(CM-3)
161  * d[4] - AR NUMBER DENSITY(CM-3)
162  * d[5] - TOTAL MASS DENSITY(GM/CM3) [includes d[8] in td7d]
163  * d[6] - H NUMBER DENSITY(CM-3)
164  * d[7] - N NUMBER DENSITY(CM-3)
165  * d[8] - Anomalous oxygen NUMBER DENSITY(CM-3)
166  * t[0] - EXOSPHERIC TEMPERATURE
167  * t[1] - TEMPERATURE AT ALT
168  *
169  *
170  * O, H, and N are set to zero below 72.5 km
171  *
172  * t[0], Exospheric temperature, is set to global average for
173  * altitudes below 120 km. The 120 km gradient is left at global
174  * average value for altitudes below 72 km.
175  *
176  * d[5], TOTAL MASS DENSITY, is NOT the same for subroutines GTD7
177  * and GTD7D
178  *
179  * SUBROUTINE GTD7 -- d[5] is the sum of the mass densities of the
180  * species labeled by indices 0-4 and 6-7 in output variable d.
181  * This includes He, O, N2, O2, Ar, H, and N but does NOT include
182  * anomalous oxygen (species index 8).
183  *
184  * SUBROUTINE GTD7D -- d[5] is the "effective total mass density
185  * for drag" and is the sum of the mass densities of all species
186  * in this model, INCLUDING anomalous oxygen.
187  */
